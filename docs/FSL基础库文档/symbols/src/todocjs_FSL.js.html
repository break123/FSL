 70      * @param {String|Element|null} selector 选择器 可以是id、class、node、tagname、不带尖括号的tagname和*
 71      * @param {Element|NodeList|Array} [limitNode] DOM范围。限制节点范围的节点元素或元素数组或元素结合，一般是selector的父节点,可选参数

 85     FS.$$ = function(selector, limitNode) {
 86         // 如果传入的参数为null、undefined或nodeType, 则返回它本身
 87         if (selector === null || typeof(selector) === 'undefined') {
 88             return selector;
 89         }
 90 
 91         var elements = [];
 92         var match = {
 93             ID: /#((?:[\w\u00c0-\uFFFF-]|\\.)+)/,
 94             CLASS: /\.((?:[\w\u00c0-\uFFFF-]|\\.)+)/,
 95             TAG: /^<(\w+)\s*\/?>(?:<\/\1>)?$/,
 96             UTAG: /^(\w+)\s*/
 97         };
 98 
 99         // 如果传入的是一个ELEMENTS元素
100         if (selector.nodeType === 1 || selector.nodeType === 9) {
101             elements.push(selector);
102             return elements;
103         }
104 
105         if (typeof selector === 'string') {
106             // 执行正则表达式，得到一个二维数组
107             var match_id = match.ID.exec(selector),
108                 match_class = match.CLASS.exec(selector),
109                 match_tag = match.TAG.exec(selector),
110                 match_utag = match.UTAG.exec(selector),
111                 element = null,
112                 allElement = null;
113 
114             // 如果limitNode存在，则筛选出limitNode子节点中的符合条件的elements，并将其存入limitNodes数组中
115             if (limitNode != null) {
116                 var elementsLimit = [],
117                     limit = [];
118 
119                 // 如果limitNode是节点元素、元素集合、节点数组
120                 // 分别将所有的限制节点放入limit数组中，下面就统一处理limit数组就可以了                
121                 if (limitNode.nodeType) {
122                     limit.push(limitNode);
123                 } else if (limitNode.item) {
124                     limit = FS.makeArray(limitNode);
125                 } else if (FS.isArray(limitNode)) {
126                     // 取得限制节点数组的长度
127                     var limitLeng = limitNode.length - 1;
128                     // 循环所有限制节点，将元素节点放入limit数组
129                     for (; limitLeng >= 0; limitLeng--) {
130                         if (limitNode[limitLeng].nodeType === 1) {
131                             limit.push(limitNode[limitLeng]);
132                         }
133                     }
134                 } else {
135                     FS.error('limitNode参数类型不对，应传入节点元素或节点元素数组或元素结合!');
136                 }
137 
138                 // 取出所有限制节点的所有子节点，存入limitNodes数组中，下面就只需要操作limitNodes数组
139                 var limitLength = limit.length - 1,
140                     limitNodes = [];
141                 for (; limitLength >= 0; limitLength--) {
142                     var limitElements = FS.getAll(limit[limitLength]),
143                         limitElementsLeng = limitElements.length - 1;
144                     for (; limitElementsLeng >= 0; limitElementsLeng--) {
145                         limitNodes.push(limitElements[limitElementsLeng]);
146                     }
147                 };
148             }
149 
150             // 选择出匹配selector的所有节点
151             // 如果浏览器支持document.querySelectorAll(IE8及以上、其他标准浏览器),则直接使用querySelectorAll
152             if (document.querySelectorAll) {
153                 elements = document.querySelectorAll(selector);
154             } else {
155                 // 不支持querySelectorAll的浏览器（主要是IE6、IE7）
156                 if (match_id) {
157                     element = document.getElementById(match_id[1]);
158                     // 如果找到了节点，则将其插入至elements数组
159                     if (element) {
160                         elements.push(element);
161                     }
162                 } else if (match_tag || match_utag) {
163                     if (match_tag) {
164                         elements = document.getElementsByTagName(match_tag[1]);
165                     } else {
166                         elements = document.getElementsByTagName(match_utag[1]);
167                     }
168                 } else if (match_class) {
169                     elements = FS.getElementsByClass(match_class[0]);
170                 } else if (selector === '*') {
171                     elements = FS.getAll();
172                 }
173             }
174 
175             // 如果limitNode存在，并且传入的选择器是TagName或class选择器，则筛选出limitNode的子节点中的符合条件的所有elements
176             if (limitNode != null && !match_id && selector !== '*') {
177                 var eleLength = elements.length;
178                 for (var k = 0; k < eleLength; k++) {
179                     // 如果limitNodes数组中包含匹配出来的elements,则插入elementsLimit数组
180                     if (limitNodes.contains(elements[k])) {
181                         elementsLimit.push(elements[k]);
182                     }
183                 }
184                 return elementsLimit;
185             }
186 
187             // 统一处理元素集合，如果elements是一个元素集合，用makeArray将其转化为数组。
188             if (elements.item) {
189                 return FS.makeArray(elements);
190             }
191             // 取得的不是元素集合，而且Array类型
192             return elements;
193         }
194     }

299     FS.getElementsByClass = function(searchClass) {
300         // 如果浏览器环境支持querySelectorAll,则使用querySelectorAll，因为它是DOM跟JavaScript的一个接口方法，速度会非常快
301         if (typeof document.querySelectorAll !== 'undefined') {
302             var classElements = document.querySelectorAll(searchClass);
303             return FS.makeArray(classElements);
304         } else {
305             // 先找出所有的节点
306             var allElements = document.getElementsByTagName('*');
307             // 去掉前面的'.'
308             var searchClass = searchClass.substring(1);
309             // 匹配class样式
310             var hasReg = new RegExp('(\\s|^)' + searchClass + '(\\s|$)'),
311                 length = allElements.length,
312                 classElements = [];
313             for (var i = 0; i < length; i++) {
314                 var classnames = allElements[i].className,
315                     hasClass = classnames.search(hasReg);
316                 // 如果hasClass===-1,则未匹配到
317                 if (hasClass !== -1) {
318                     classElements.push(allElements[i]);
319                 }
320             }
321             return classElements;
322         }
323     }

481     /**
482      * 移除节点或节点数组
483      * @param {Element|NodeList|Array} arguments 需要移除的节点，多个节点用逗号隔开
484      * @example  FS.remove(FS(options.maskId)[0],FS('#iframe'+maskId)[0]);
485      * @example  FS.remove(FS('select'),FS('#iframe'+maskId)[0]);
486      * @example  FS.remove(document.images);
487      */
488     FS.remove = function() {
489         try {
490             var arg = arguments,
491                 length = arg.length - 1;
492             for (; length >= 0; length--) {
493                 // 如果参数是节点数组
494                 if (arg[length] == null) {
495                     continue;
496                 } else if (FS.isArray(arg[length]) || arg[length].item) {
497                     var leng = arg[length].length - 1;
498                     for (; leng >= 0; leng--) {
499                         var argArr = arg[length][leng];
500                         if (FS.notNode(argArr)) {
501                             // 如果传入的不是node，继续下一个节点元素
502                             continue;
503                         }
504                         // 先找出父节点，再移除该节点
505                         var parentNode = argArr.parentNode;
506                         parentNode.removeChild(argArr);
507                     }
508                 } else if (FS.notNode(arg[length])) {
509                     // 如果传入的不是node，继续下一个节点元素
510                     continue;
511                 } else {
512                     var parentNode = arg[length].parentNode;
513                     parentNode.removeChild(arg[length]);
514                 }
515             }
516         } catch (msg) {
517             FS.error('FS.remove方法出现异常' + msg);
518         }
